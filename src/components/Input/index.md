### Install

#### NPM

```js
import 'zendesign/dist/static/css/common.css';
import 'zendesign/dist/js/vendor'
import Input from 'zendesign/dist/js/input.js'
import 'zendesign/dist/static/css/input.css'
```

```js
function App() {
  return (
    <>
      <Input />
    </>
  )
}
```
