### Install

#### NPM

```js

import "zendesign";
import Autocomplete from "zendesign/dist/js/Autocomplete.js";
import "zendesign/dist/static/css/Autocomplete.css";
```

```js
function App() {
  return (
    <>
      <Autocomplete options={["a", "b", "c"]} />
    </>
  );
}
```
